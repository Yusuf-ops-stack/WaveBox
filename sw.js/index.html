// sw.js const CACHE_NAME = 'wavebox-cache-v1'; const urlsToCache = [ '/', '/index.html', '/favIcon.ico', '/manifest.json', 'https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.4.0/css/all.min.css' ]; self.addEventListener('install', e => { e.waitUntil( caches.open(CACHE_NAME).then(cache => cache.addAll(urlsToCache)) ); }); self.addEventListener('fetch', e => { e.respondWith( caches.match(e.request).then(response => response || fetch(e.request)) ); });
